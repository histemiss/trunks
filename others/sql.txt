create database trunk;
use database trunk;

create table file_info(
	id BIGINT NOT NULL AUTO_INCREMENT, 
	user_id BIGINT,
	name CHAR(255),
	path CHAR(255),
	PRIMARY KEY (ID), UNIQUE(path)
);

create table user_info (
	ID BIGINT NOT NULL AUTO_INCREMENT,
	name CHAR(255),
	password CHAR(255),
	PRIMARY KEY(ID),UNIQUE(name)
);
